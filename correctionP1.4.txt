----------------Commencez par écouter les événements click depuis l'élément #parent. Puis affichez le nombre de clics dans l'élément #parent-count---------------

const parent = document.getElementById("parent");		// On récupère l'élément 'parent' qui correspond au cadre entier
const parentCount = document.getElementById("parent-count");	// On récupère l'élément 'parent-count' qui correspond au compteur "Parent"
let parentCountValue = 0;					// On créer une variable qui correspond à la valeur en 'number' du compteur et on l'initialise à 0
parent.addEventListener("click", function(){			// On créer un EventListener qui écoutera les clics sur le cadre entier ('parent') ; A chaque clic on executera la fonction suivante:
 parentCountValue++;						// La valeur de parentCountValue augmente de 1
 parentCount.textContent = parentCountValue;			// On affiche la valeur de parentCountValue dans l'élément 'parent-count'
});

----------------Faites la même chose mais avec l'élément #child---------------

const child = document.getElementById("child");
const childCount = document.getElementById("child-count");
let childCountValue = 0;
child.addEventListener("click", function(){
 childCountValue++;
 childCount.textContent = childCountValue;
});


----------------Maintenant, faites de sorte à ce que lorsque vous cliquez sur l'enfant, seul le compteur de l'enfant se mette à jour---------------

child.addEventListener("click", function(event){		// On ajoute le paramètre 'event' afin de pouvoir agir dessus
 event.stopPropagation();					// On précise que l'on veut que cet 'event' ne se propage pas à son parent
 childCountValue++;
 childCount.textContent = childCountValue;
});


----------------Evitez qu'un clic sur le lien ne vous fasse changer de page: supprimez ce comportement par défaut.---------------

child.addEventListener("click", function(event){
 event.stopPropagation();
 event.preventDefault();
 childCountValue++;
 childCount.textContent = childCountValue;
});
