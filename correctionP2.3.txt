------- JS --------------

const code = document.getElementById("code");			// on définit la constante code qui est l'input du code à saisir
const codeValide = document.getElementById("code-validation"); 	// on définit la constante code qui est le texte qui affichera si le code est valide

code.addEventListener("input", function (code) { 		// On créer un "écouteur d'événement sur chaque input qui va vérifier si le texte saisie dans l'input "code" commence bien par "CODE-"
  const codeRgex = RegExp(/^CODE-/); 				// On créer une règle RegExp avec la string obligatoire en début d'input
  let codeResult = codeRgex.test(code.target.value); 		// On créer une variable qui va vérifier si le paramètre (code) de notre fonction répond à la règle fixée par la RegExp
  const submit = document.getElementById("submit-btn"); 	// On a besoin de faire appel à notre bouton de soumission pour la règle suivante 
  submit.setAttribute("disabled", "true"); 			// On ajoute par défaut, l'attribut "disabled" à notre boutton de soumssion, par défaut, il est grisé
  if (codeResult === true) { 					// Si le résultat de notre constante "code" commence par "CODE-"
    codeValide.innerHTML = "Code valide"; 			// Alors un texte "Code valide" s'affiche dans la zone "code-validation"
    submit.removeAttribute("disabled"); 			// et on supprime l'attribut "disabled", le bouton n'est plus grisé.
  } else { // Sinon
    codeValide.innerHTML = "Code invalide"; 			// Le texte affiche "Code invalide", le bouton reste grisé
  }
});
